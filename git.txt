git init // Inicia gits

git config --global user.name "nombre de usuario"
git config --global user.email "email de usuario"

git clone origin -b [rama] [id de commit] [url] // Clona una rama

git commit -m "[comentario(remove,add,fix,etc)]"

git status // Estado de archivo locales

git push [nombre] // Sube cambios a GitHub

git remote add [nombre de repositorio] [url]

git branch // Muestra las ramas y marca en la que estoy parado
git branch [nombre] / Crea una rama
git branch -d [nombre] // Elimina una rama

git push --set-upstream origin [nombre] // Publica una rama en GitHub por primera vez

git ferch // Actualiza el estado de la rama local con la rama de la nube

git pull // Baja los cambios de la nube a la rama local
git pull origin [nombre] // Lo mismo pero nombrando

git merge origin [nombre] // Unifica las ramas
